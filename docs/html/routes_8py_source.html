61def autenticar_com_kerberos(
62 usuario,
63 senha,
64 usar_rede=True,
65 cliente_tcp=None
66):
67 """
68 @brief Executa o fluxo completo Cliente-AS-TGS-Portal.
69
70 @param usuario Usuario informado no login.
71 @param senha Senha usada localmente para derivar a chave do cliente.
72 @param usar_rede Define se AS, TGS e Notas serao chamados por TCP.
73 @param cliente_tcp Cliente TCP injetavel para testes.
74 @return Dados de sessao Kerberos mantidos no servidor Flask.
75 @throws ValueError Quando qualquer validacao Kerberos falha.
76 """
77 logs = []
78 registrar_etapa(logs, "[CLIENTE] Senha informada localmente pelo usuario.")
79 registrar_etapa(logs, "[CLIENTE] Solicitando autenticacao ao AS.")
80
81 cliente_tcp = cliente_tcp or ClienteKerberosTCP()
82 if usar_rede:
83 parametros_as = cliente_tcp.solicitar_parametros_as(usuario)
84 else:
85 parametros_as = criar_desafio_as(usuario)
86
87 if parametros_as.get("iteracoes_kdf") != ITERACOES_PBKDF2:
88 raise ValueError("Parametros KDF recebidos do AS sao invalidos.")
89
90 chave_derivada = derivar_chave_senha(senha, parametros_as["salt"])
91 chave_cliente = obter_chave_autenticacao_as(chave_derivada)
92 prova = gerar_prova_as(
93 chave_cliente,
94 usuario,
95 parametros_as["desafio"],
96 )
97 if usar_rede:
98 resposta_as_criptografada = cliente_tcp.enviar_prova_as(
99 usuario,
100 parametros_as["desafio"],
101 prova,
102 )
103 else:
104 resposta_as_criptografada = autenticar_no_as_com_prova(
105 usuario,
106 parametros_as["desafio"],
107 prova,
108 )
109 registrar_etapa(
110 logs,
111 "[AS] Prova criptografica do usuario validada.",
112 )
113
114 registrar_etapa(logs, "[AS] TGT emitido com sucesso.")
115 registrar_etapa(logs, "[CLIENTE] Chave derivada com PBKDF2-HMAC-SHA256.")
116
117 resposta_as = descriptografar_json(chave_cliente, resposta_as_criptografada)
118 chave_sessao_cliente_tgs = resposta_as["chave_sessao_cliente_tgs"]
119 autenticador_tgs = criar_autenticador(usuario, chave_sessao_cliente_tgs)
120 registrar_etapa(logs, "[CLIENTE] Autenticador Cliente-TGS criado.")
121
122 if usar_rede:
123 resposta_tgs = cliente_tcp.solicitar_ticket_servico(
124 usuario,
125 "notas",
126 resposta_as["tgt"],
127 autenticador_tgs,
128 )
129 else:
130 resposta_tgs = emitir_ticket_servico(
131 usuario=usuario,
132 servico="notas",
133 tgt_criptografado=resposta_as["tgt"],
134 autenticador_criptografado=autenticador_tgs,
135 )
136 registrar_etapa(logs, "[TGS] TGT e autenticador Cliente-TGS validados.")
137 registrar_etapa(logs, "[TGS] Service Ticket para o Portal de Notas emitido.")
138
139 dados_cliente = descriptografar_json(
140 base64_para_bytes(chave_sessao_cliente_tgs),
141 resposta_tgs["resposta_cliente"],
142 )
143 chave_sessao_servico = dados_cliente["chave_sessao_cliente_servico"]
144
145 nonce_portal = secrets.token_hex(16)
146 autenticador_portal = criar_autenticador(
147 usuario,
148 chave_sessao_servico,
149 nonce=nonce_portal,
150 )
151 dados_autenticador = abrir_autenticador(
152 chave_sessao_servico,
153 autenticador_portal,
154 )
155 registrar_etapa(logs, "[CLIENTE] Service Ticket e autenticador enviados ao Portal.")
156
157 if usar_rede:
158 confirmacao_portal = cliente_tcp.autenticar_portal(
159 resposta_tgs["ticket_servico"],
160 autenticador_portal,
161 )
162 else:
163 confirmacao_portal = autenticar_portal_notas(
164 resposta_tgs["ticket_servico"],
165 autenticador_portal,
166 )
167 registrar_etapa(logs, "[PORTAL] Service Ticket validado.")
168 registrar_etapa(logs, "[PORTAL] Autenticador Cliente-Servico validado.")
169
170 validar_confirmacao_portal(
171 chave_sessao_servico,
172 confirmacao_portal,
173 dados_autenticador["timestamp"],
174 nonce_portal,
175 )
176 registrar_etapa(logs, "[PORTAL] Autenticacao mutua concluida.")
177
178 perfil = obter_perfil_usuario(usuario) if not usar_rede else None
179 resultado = {
180 "usuario": usuario,
181 "perfil": perfil,
182 "ticket_servico": resposta_tgs["ticket_servico"],
183 "chave_sessao_servico": chave_sessao_servico,
184 "portal_autenticado": True,
185 "usar_rede": usar_rede,
186 "cliente_tcp": cliente_tcp,
187 "logs": logs,
188 }
189
190 if usar_rede:
191 painel = executar_operacao_kerberos(resultado, "carregar_painel")
192 perfil = painel["perfil"]
193 resultado["perfil"] = perfil
194
195 registrar_etapa(logs, f"[PORTAL] Acesso autorizado para perfil {perfil}.")
196 return resultado

216 """

237def executar_operacao_kerberos(dados_sessao, acao, dados=None):
238 """
239 @brief Executa uma operacao do Portal usando Kerberos fim a fim.
240
241 @param dados_sessao Sessao Kerberos criada no login.
242 @param acao Operacao do Portal de Notas.
243 @param dados Dados especificos da operacao.
244 @return Resultado validado enviado pelo servico.
245
246 A funcao cria uma requisicao cifrada, calcula seu hash, prende esse hash no
247 autenticador e valida a resposta do Portal com timestamp e nonce esperados.
248 """
249 usuario = dados_sessao["usuario"]
250 chave_sessao = dados_sessao["chave_sessao_servico"]
251 nonce_operacao = secrets.token_hex(16)
252 requisicao = {
253 "usuario": usuario,
254 "acao": acao,
255 "dados": dados or {},
256 "nonce": nonce_operacao,
257 }
258 hash_requisicao = calcular_hash_requisicao(requisicao)
259 requisicao_criptografada = criptografar_json(
260 base64_para_bytes(chave_sessao),
261 requisicao,
262 )
263 autenticador = criar_autenticador(
264 usuario,
265 chave_sessao,
266 nonce=nonce_operacao,
267 acao=acao,
268 hash_requisicao=hash_requisicao,
269 )
270 dados_autenticador = abrir_autenticador(chave_sessao, autenticador)
271
272 registrar_etapa(
273 dados_sessao["logs"],
274 f"[CLIENTE] Autenticador criado para a operacao {acao}.",
275 )
276 if dados_sessao.get("usar_rede"):
277 cliente_tcp = dados_sessao.get("cliente_tcp") or ClienteKerberosTCP()
278 resposta_criptografada = cliente_tcp.executar_operacao(
279 dados_sessao["ticket_servico"],
280 autenticador,
281 requisicao_criptografada,
282 )
283 else:
284 resposta_criptografada = processar_operacao_portal(
285 dados_sessao["ticket_servico"],
286 autenticador,
287 requisicao_criptografada,
288 )
289 registrar_etapa(
290 dados_sessao["logs"],
291 f"[PORTAL] Ticket, autenticador e requisicao {acao} validados.",
292 )
293
294 resposta = validar_resposta_operacao(
295 chave_sessao,
296 resposta_criptografada,
297 acao,
298 dados_autenticador["timestamp"],
299 nonce_operacao,
300 )
301 registrar_etapa(
302 dados_sessao["logs"],
303 f"[CLIENTE] Autenticacao mutua concluida para {acao}.",
304 )
305 return resposta["resultado"]

308def create_app(usar_rede=True, cliente_tcp=None):
309 """
310 @brief Cria a aplicacao Flask do Portal de Notas.
311
312 @param usar_rede Define se as rotas usam os servidores TCP reais.

316 app = Flask(
317 __name__,
318 template_folder=str(BASE_DIR / "templates"),

321 app.secret_key = os.environ.get(

324 app.config["KERBEROS_USAR_REDE"] = usar_rede
325
326 # O cookie recebe somente este identificador. Tickets e chaves ficam no servidor.

330 def obter_sessao_kerberos():
331 """@brief Recupera a sessao Kerberos associada ao cookie Flask."""
332 id_sessao = session.get("id_sessao_kerberos")
333 if not id_sessao:
334 return None
335 return sessoes_kerberos.get(id_sessao)

337 def exigir_sessao_kerberos():
338 """@brief Recupera e valida a sessao Kerberos antes das rotas protegidas."""
339 dados_sessao = obter_sessao_kerberos()
340 if not dados_sessao:
341 return None
342 validar_sessao_portal(dados_sessao)
343 return dados_sessao
