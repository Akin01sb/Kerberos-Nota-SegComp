49 def _solicitar(self, porta, requisicao):
50 """
51 @brief Envia uma requisicao para uma porta Kerberos e retorna o resultado.
52
53 @param porta Porta TCP do servidor.
54 @param requisicao Dicionario JSON da chamada remota.
55 @return Campo `resultado` da resposta remota.
56 @throws ConnectionError Quando nao ha servidor disponivel.
57 @throws PermissionError Quando o servidor remoto negou permissao.
58 @throws ValueError Para demais erros remotos.
59 """
60 try:
61 with socket.create_connection(
62 (self.host, porta),
63 timeout=self.timeout,
64 ) as conexao:
65 conexao.settimeout(self.timeout)
66 enviar_mensagem(conexao, requisicao)
67 resposta = receber_mensagem(conexao)
68 except OSError as erro:
69 raise ConnectionError(
70 f"Nao foi possivel conectar ao servico na porta {porta}."
71 ) from erro
72
73 if resposta.get("sucesso"):
74 return resposta["resultado"]
75
76 mensagem = resposta.get("erro", "Erro remoto sem detalhes.")
77 if resposta.get("tipo_erro") == "PermissionError":
78 raise PermissionError(mensagem)
79 raise ValueError(mensagem)
