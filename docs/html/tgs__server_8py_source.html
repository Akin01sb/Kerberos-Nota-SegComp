56def _registrar_nonce_tgs(usuario, nonce, timestamp):
57 """
58 @brief Registra nonces ja usados em autenticadores Cliente-TGS.
59
60 @param usuario Usuario dono do autenticador.
61 @param nonce Valor unico informado pelo cliente.
62 @param timestamp Timestamp do autenticador validado.
63 @throws ValueError Quando o nonce esta ausente ou foi reutilizado.
64 """
65 if not nonce:
66 raise ValueError("Autenticador sem nonce.")
67
68 with BLOQUEIO_NONCES_TGS:
69 limite = timestamp_atual() - TEMPO_MAXIMO_AUTENTICADOR
70 expirados = [
71 chave
72 for chave, momento in NONCES_TGS_UTILIZADOS.items()
73 if momento < limite
74 ]
75 for chave in expirados:
76 NONCES_TGS_UTILIZADOS.pop(chave, None)
77
78 chave_nonce = (usuario, nonce)
79 if chave_nonce in NONCES_TGS_UTILIZADOS:
80 raise ValueError("Autenticador reutilizado no TGS: possivel replay.")
81
82 NONCES_TGS_UTILIZADOS[chave_nonce] = timestamp

85def validar_tgt(usuario: str, tgt_criptografado: dict) -> dict:
86 """
87 @brief Abre e valida um TGT recebido pelo TGS.
88
89 @param usuario Usuario que solicita o Service Ticket.
90 @param tgt_criptografado TGT cifrado com a chave secreta do TGS.
91 @return TGT em claro depois da validacao.
92 @throws ValueError Para TGT ausente, adulterado, expirado ou de outro usuario.
93 """
94 if not tgt_criptografado:
95 raise ValueError("TGT nao informado.")
96
97 try:
98 tgt = descriptografar_json(CHAVE_SECRETA_TGS, tgt_criptografado)
99 except Exception as erro:
100 raise ValueError("TGT invalido ou nao pode ser descriptografado.") from erro
101
102 if tgt.get("id_cliente") != usuario:
103 raise ValueError("TGT pertence a outro usuario.")
104
105 chave_sessao = tgt.get("chave_sessao_cliente_tgs")
106 if not chave_sessao:
107 raise ValueError("TGT nao contem chave de sessao Cliente-TGS.")
108
109 if ticket_expirou(tgt["timestamp_emissao"], tgt["validade_segundos"]):
110 raise ValueError("TGT expirado.")
111
112 return tgt

115def validar_autenticador(usuario: str, tgt: dict, autenticador_criptografado: dict) -> dict:
116 """
117 @brief Valida o autenticador Cliente-TGS associado a um TGT.
118
119 @param usuario Usuario esperado.
120 @param tgt TGT ja descriptografado e validado.
121 @param autenticador_criptografado Autenticador cifrado com a chave Cliente-TGS.
122 @return Autenticador em claro.
123 @throws ValueError Para autenticador invalido, expirado, futuro ou reutilizado.
124 """
125 if not autenticador_criptografado:
126 raise ValueError("Autenticador nao informado.")
127
128 chave_sessao = tgt["chave_sessao_cliente_tgs"]
129
130 try:
131 autenticador = abrir_autenticador(chave_sessao, autenticador_criptografado)
132 except Exception as erro:
133 raise ValueError("Autenticador invalido ou nao pode ser descriptografado.") from erro
134
135 if autenticador.get("usuario") != usuario:
136 raise ValueError("Usuario do autenticador diferente do usuario do TGT.")
137
138 timestamp = autenticador.get("timestamp")
139 if timestamp is None:
140 raise ValueError("Autenticador sem timestamp.")
141
142 if ticket_expirou(timestamp, TEMPO_MAXIMO_AUTENTICADOR):
143 raise ValueError("Autenticador expirado.")
144
145 if timestamp > timestamp_atual() + TEMPO_MAXIMO_AUTENTICADOR:
146 raise ValueError("Autenticador com timestamp invalido.")
147
148 _registrar_nonce_tgs(usuario, autenticador.get("nonce"), timestamp)
149 return autenticador

152def emitir_ticket_servico(
153 usuario: str,
154 servico: str,
155 tgt_criptografado: dict,
156 autenticador_criptografado: dict
157) -> dict:
158 """
159 @brief Emite Service Ticket e chave Cliente-Servico.
160
161 @param usuario Usuario autenticado no AS.
162 @param servico Servico protegido solicitado.
163 @param tgt_criptografado Ticket Granting Ticket transportado pelo cliente.
164 @param autenticador_criptografado Prova recente cifrada com a chave Cliente-TGS.
165 @return Pacote com Service Ticket cifrado e resposta cifrada para o cliente.
166 """
167 chave_servico = obter_chave_servico(servico)
168 tgt = validar_tgt(usuario, tgt_criptografado)
169 validar_autenticador(usuario, tgt, autenticador_criptografado)
170
171 chave_sessao_cliente_servico = gerar_chave_simetrica()
172 chave_sessao_cliente_servico_base64 = bytes_para_base64(chave_sessao_cliente_servico)
173
174 ticket_servico = criar_ticket_servico(
175 usuario=usuario,
176 servico=servico,
177 chave_sessao_cliente_servico_base64=chave_sessao_cliente_servico_base64
178 )
179
180 ticket_servico_criptografado = criptografar_json(chave_servico, ticket_servico)
181 chave_cliente_tgs = base64_para_bytes(tgt["chave_sessao_cliente_tgs"])
182
183 resposta_cliente = criptografar_json(
184 chave_cliente_tgs,
185 {
186 "usuario": usuario,
187 "servico": servico,
188 "chave_sessao_cliente_servico": chave_sessao_cliente_servico_base64,
189 "timestamp_emissao": ticket_servico["timestamp_emissao"],
190 "timestamp_expiracao": ticket_servico["timestamp_expiracao"],
191 "nonce_ticket": ticket_servico["nonce"]
192 }
193 )
194
195 return {
196 "servico": servico,
197 "ticket_servico": ticket_servico_criptografado,
198 "resposta_cliente": resposta_cliente
199 }

202def abrir_ticket_servico(servico: str, ticket_servico_criptografado: dict) -> dict:
203 """
204 @brief Abre e valida um Service Ticket no servico protegido.
205
206 @param servico Nome do servico que esta recebendo o ticket.
207 @param ticket_servico_criptografado Ticket cifrado com a chave do servico.
208 @return Ticket de servico em claro.
209 @throws ValueError Quando o ticket e invalido, de outro servico ou expirou.
210 """
211 chave_servico = obter_chave_servico(servico)
212
213 try:
214 ticket_servico = descriptografar_json(chave_servico, ticket_servico_criptografado)
215 except Exception as erro:
216 raise ValueError("Ticket de servico invalido ou nao pode ser descriptografado.") from erro
217
218 if ticket_servico.get("servico") != servico:
219 raise ValueError("Ticket foi emitido para outro servico.")
220
221 if ticket_expirou_por_timestamp(ticket_servico["timestamp_expiracao"]):
222 raise ValueError("Ticket de servico expirado.")
223
224 return ticket_servico
