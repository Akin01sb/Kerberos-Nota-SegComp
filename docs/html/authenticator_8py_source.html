21def criar_autenticador(
22 usuario: str,
23 chave_sessao_base64: str,
24 nonce: str | None = None,

28 """
29 @brief Cria autenticador cifrado com uma chave de sessao.
30
31 @param usuario Usuario que apresenta o ticket.
32 @param chave_sessao_base64 Chave de sessao em Base64.
33 @param nonce Valor unico contra replay; se ausente, e gerado.
34 @param acao Operacao esperada no servico, quando aplicavel.
35 @param hash_requisicao Hash da requisicao protegida, quando aplicavel.
36 @return Pacote AES-GCM contendo os dados do autenticador.
37 """
38 chave_sessao = base64_para_bytes(chave_sessao_base64)
39
40 dados = {
41 "usuario": usuario,
42 "timestamp": timestamp_atual(),
43 "nonce": nonce or uuid.uuid4().hex
44 }
45
46 if acao:
47 dados["acao"] = acao
48
49 if hash_requisicao:
50 dados["hash_requisicao"] = hash_requisicao
51
52 return criptografar_json(chave_sessao, dados)
