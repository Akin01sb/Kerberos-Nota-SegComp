82def _normalizar_nota(valor):
83 """
84 @brief Converte uma nota textual ou numerica para float valido.
85
86 @param valor Valor recebido do formulario ou teste.
87 @return Nota normalizada entre 0 e 10.
88 @throws ValueError Quando o valor nao e numerico ou esta fora da faixa.
89 """
90 try:
91 nota = float(str(valor).replace(",", "."))
92 except (TypeError, ValueError) as erro:
93 raise ValueError("Informe uma nota numerica.") from erro
94
95 if not 0 <= nota <= 10:
96 raise ValueError("A nota deve estar entre 0 e 10.")
97
98 return nota

102 professor,
103 aluno,
104 disciplina,
105 nota,
106 observacao="",
107 perfil=PERFIL_PROFESSOR
108):
109 """
110 @brief Cria uma nota para um aluno.
111
112 @param professor Usuario professor autenticado.
113 @param aluno Aluno dono da nota.
114 @param disciplina Nome da disciplina.
115 @param nota Valor numerico entre 0 e 10.
116 @param observacao Texto opcional.
117 @param perfil Perfil do usuario que esta executando a acao.
118 @return Nota salva no repositorio.
119 """
120 _validar_professor(perfil)
121
122 if not professor:
123 raise ValueError("Professor nao informado.")
124
125 aluno = (aluno or "").strip()

131
132 if not disciplina:
133 raise ValueError("Disciplina nao informada.")
134

137 "nota": _normalizar_nota(nota),

143def criar_notas(
144 professor,
145 aluno,
146 notas,
147 perfil=PERFIL_PROFESSOR
148):
149 """
150 @brief Cria varias notas para um aluno em uma unica operacao.
151
152 @param professor Usuario professor autenticado.
153 @param aluno Aluno dono das notas.
154 @param notas Lista de dicionarios com disciplina, nota e observacao.
155 @param perfil Perfil do usuario que esta executando a acao.
156 @return Lista de notas salvas.
157 """
158 _validar_professor(perfil)
159 if not professor:
160 raise ValueError("Professor nao informado.")
161
162 aluno = (aluno or "").strip()
163 if aluno not in listar_alunos():
164 raise ValueError("Aluno nao encontrado.")
165 if not isinstance(notas, list) or not notas:
166 raise ValueError("Informe pelo menos uma nota.")
167
168 notas_validadas = []
169 for item in notas:
170 disciplina = (item.get("disciplina") or "").strip()
171 if not disciplina:
172 raise ValueError("Disciplina nao informada.")
173 notas_validadas.append({
174 "disciplina": disciplina,
175 "nota": _normalizar_nota(item.get("nota")),
176 "observacao": (item.get("observacao") or "").strip(),
177 "professor": professor,
178 })
179
180 return [
181 adicionar_nota_usuario(aluno, item)
182 for item in notas_validadas
183 ]
