49def criar_usuario(nome_usuario: str, senha: str, perfil: str = "aluno") -> None:
50 """
51 @brief Cadastra um usuario com senha derivada por KDF.
52
53 @param nome_usuario Nome do usuario.
54 @param senha Senha em texto claro apenas durante o cadastro.
55 @param perfil Perfil `aluno` ou `professor`.
56 @throws ValueError Quando o perfil informado e invalido.
57 """
58 dados= carregar_usuarios()
59
60 if nome_usuario in dados["usuarios"]:
61 print("Usuario já existe.")
62 return
63
64 if perfil not in {"professor", "aluno"}:
65 raise ValueError("Perfil deve ser professor ou aluno.")
66
67 salt = gerar_salt()
68 chave = derivar_chave_senha(senha, salt)
69 verificador = gerar_verificador_chave(chave)
70
71 dados["usuarios"][nome_usuario] = {
72 "salt": salt,
73 "verificador": verificador,
74 "perfil": perfil,
75 }
76
77 salvar_usuarios(dados)
78
79 print(f"Usuario '{nome_usuario}' criado com sucesso.")
