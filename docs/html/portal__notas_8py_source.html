43def calcular_hash_requisicao(requisicao):
44 """
45 @brief Calcula hash canonico de uma requisicao protegida.
46
47 @param requisicao Dicionario com usuario, acao, dados e nonce.
48 @return Digest SHA-256 hexadecimal usado no autenticador.
49 """
50 texto = json.dumps(
51 requisicao,
52 ensure_ascii=False,
53 sort_keys=True,
54 separators=(",", ":"),
55 )
56 return hashlib.sha256(texto.encode("utf-8")).hexdigest()

73def _registrar_nonce(usuario, nonce, timestamp):
74 """
75 @brief Registra nonce de autenticador Cliente-Servico contra replay.
76
77 @param usuario Usuario dono do autenticador.
78 @param nonce Valor unico enviado pelo cliente.
79 @param timestamp Timestamp do autenticador validado.
80 @throws ValueError Quando o nonce esta ausente ou ja foi usado.
81 """
82 if not nonce:
83 raise ValueError("Autenticador Cliente-Servico sem nonce.")
84
85 with BLOQUEIO_NONCES:
86 limite = timestamp_atual() - TEMPO_MAXIMO_AUTENTICADOR
87 expirados = [
88 chave
89 for chave, momento in NONCES_UTILIZADOS.items()
90 if momento < limite
91 ]
92 for chave in expirados:
93 NONCES_UTILIZADOS.pop(chave, None)
94
95 chave_nonce = (usuario, nonce)
96 if chave_nonce in NONCES_UTILIZADOS:
97 raise ValueError("Autenticador reutilizado: possivel ataque de replay.")
98
99 NONCES_UTILIZADOS[chave_nonce] = timestamp

102def _validar_autenticador_portal(ticket, autenticador_criptografado):
103 """
104 @brief Valida autenticador Cliente-Servico usando a chave do ticket.
105
106 @param ticket Service Ticket ja aberto.
107 @param autenticador_criptografado Autenticador cifrado pelo cliente.
108 @return Autenticador em claro.
109 @throws ValueError Para autenticador invalido, expirado, futuro ou reutilizado.
110 """
111 chave_sessao_base64 = ticket["chave_sessao_cliente_servico"]
112
113 try:
114 autenticador = abrir_autenticador(
115 chave_sessao_base64,
116 autenticador_criptografado,
117 )
118 except Exception as erro:
119 raise ValueError("Autenticador Cliente-Servico invalido.") from erro
120
121 if autenticador.get("usuario") != ticket.get("usuario"):
122 raise ValueError("Autenticador pertence a outro usuario.")
123
124 timestamp = autenticador.get("timestamp")
125 if timestamp is None:
126 raise ValueError("Autenticador Cliente-Servico sem timestamp.")
127
128 if ticket_expirou(timestamp, TEMPO_MAXIMO_AUTENTICADOR):
129 raise ValueError("Autenticador Cliente-Servico expirado.")
130
131 if timestamp > timestamp_atual() + TEMPO_MAXIMO_AUTENTICADOR:
132 raise ValueError("Autenticador Cliente-Servico com timestamp invalido.")
133
134 _registrar_nonce(ticket["usuario"], autenticador.get("nonce"), timestamp)
135 return autenticador

263def processar_operacao_portal(
264 ticket_servico_criptografado,
265 autenticador_criptografado,
266 requisicao_criptografada
267):
268 """
269 @brief Processa uma operacao de notas protegida por Kerberos.
270
271 @param ticket_servico_criptografado Service Ticket do usuario.
272 @param autenticador_criptografado Autenticador amarrado a acao e requisicao.
273 @param requisicao_criptografada Requisicao cifrada com a chave Cliente-Servico.
274 @return Resposta cifrada com o resultado da operacao.
275 @throws ValueError Quando ticket, autenticador, nonce, acao ou hash falham.
276 """
277 ticket = validar_ticket_portal(ticket_servico_criptografado)
278 autenticador = _validar_autenticador_portal(
279 ticket,
280 autenticador_criptografado,
281 )
282 chave_sessao_base64 = ticket["chave_sessao_cliente_servico"]
283 chave_sessao = base64_para_bytes(chave_sessao_base64)
284
285 try:
286 requisicao = descriptografar_json(
287 chave_sessao,
288 requisicao_criptografada,
289 )
290 except Exception as erro:
291 raise ValueError("Requisicao protegida invalida ou adulterada.") from erro
292
293 if requisicao.get("usuario") != ticket["usuario"]:
294 raise ValueError("Usuario da requisicao diferente do Service Ticket.")
295
296 if requisicao.get("nonce") != autenticador.get("nonce"):
297 raise ValueError("Nonce da requisicao diferente do autenticador.")
298
299 acao = requisicao.get("acao")
300 if acao != autenticador.get("acao"):
301 raise ValueError("Acao da requisicao diferente do autenticador.")
302
303 hash_calculado = calcular_hash_requisicao(requisicao)
304 if not hmac.compare_digest(
305 hash_calculado,
306 autenticador.get("hash_requisicao", ""),
307 ):
308 raise ValueError("Requisicao nao corresponde ao autenticador.")
309
310 resultado = _executar_acao(
311 ticket["usuario"],
312 acao,
313 requisicao.get("dados", {}),
314 )
315
316 resposta = {
317 "status": "operacao_concluida",
318 "acao": acao,
319 "timestamp_resposta": autenticador["timestamp"] + 1,
320 "nonce_autenticador": autenticador["nonce"],
321 "resultado": resultado,
322 }
323 return criptografar_json(chave_sessao, resposta)
