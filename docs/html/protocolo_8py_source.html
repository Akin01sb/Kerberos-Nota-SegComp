20def _receber_exatamente(conexao, quantidade):
21 """
22 @brief Le exatamente a quantidade de bytes solicitada.
23
24 @param conexao Socket conectado.
25 @param quantidade Numero de bytes esperados.
26 @return Bytes recebidos.
27 @throws ConnectionError Quando a conexao encerra antes da mensagem completa.
28 """
29 partes = []
30 restante = quantidade
31
32 while restante:
33 parte = conexao.recv(restante)
34 if not parte:
35 raise ConnectionError("Conexao encerrada antes do fim da mensagem.")
36 partes.append(parte)
37 restante -= len(parte)
38
39 return b"".join(partes)

42def enviar_mensagem(conexao, dados):
43 """
44 @brief Envia um dicionario JSON com cabecalho de tamanho.
45
46 @param conexao Socket conectado.
47 @param dados Dicionario serializavel em JSON.
48 @throws ValueError Quando a mensagem excede o limite configurado.
49 """
50 conteudo = json.dumps(
51 dados,
52 ensure_ascii=False,
53 separators=(",", ":"),
54 ).encode("utf-8")
55 if len(conteudo) > TAMANHO_MAXIMO_MENSAGEM:
56 raise ValueError("Mensagem de rede muito grande.")
57
58 cabecalho = struct.pack(FORMATO_CABECALHO, len(conteudo))
59 conexao.sendall(cabecalho + conteudo)

62def receber_mensagem(conexao):
63 """
64 @brief Recebe e decodifica uma mensagem JSON completa.
65
66 @param conexao Socket conectado.
67 @return Dicionario decodificado do JSON recebido.
68 @throws ValueError Quando o tamanho anunciado e invalido.
69 """
70 cabecalho = _receber_exatamente(conexao, TAMANHO_CABECALHO)
71 tamanho = struct.unpack(FORMATO_CABECALHO, cabecalho)[0]
72 if tamanho <= 0 or tamanho > TAMANHO_MAXIMO_MENSAGEM:
73 raise ValueError("Tamanho de mensagem de rede invalido.")
74
75 conteudo = _receber_exatamente(conexao, tamanho)
76 return json.loads(conteudo.decode("utf-8"))
