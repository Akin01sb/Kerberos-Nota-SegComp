31def gerar_chave_simetrica() -> bytes:
32 """

36 @brief Gera uma chave simetrica de 256 bits.

39 Usa `os.urandom` para produzir 32 bytes aleatorios. Essas chaves sao usadas

42
43 Parametros:

45
46 Valor retornado:

132def criptografar_json(chave: bytes, dados: dict) -> dict:
133 """
134 ***************************************************************************
135 Funcao: criptografar_json
136
137 @brief Cifra um dicionario JSON com AES-GCM.
138
139 Descricao:
140 Serializa um dicionario em JSON, gera um nonce de 12 bytes e cifra os dados
141 com AES-GCM. O retorno contem nonce e ciphertext em Base64, formato usado
142 nos tickets, autenticadores e respostas do protocolo.
143
144 Parametros:
145 @param chave Chave AES de 32 bytes.
146 @param dados Dicionario serializavel em JSON.
147
148 Valor retornado:
149 @return Dicionario com `nonce` e `ciphertext` em Base64.
150
151 Assertiva de entrada:
152 @pre A chave deve ter tamanho aceito pelo AES-GCM.
153 @pre Os dados devem ser serializaveis em JSON.
154
155 Assertiva de saida:
156 @post Retorna pacote cifrado e autenticado.
157
158 Excecoes:
159 @throws TypeError Pode ocorrer se os dados nao forem serializaveis.
160 @throws Exception Pode propagar erros da biblioteca criptografica.
161
162 Observacoes:
163 AES-GCM fornece confidencialidade e deteccao de adulteracao. Nao ha dados
164 autenticados adicionais associados nesta implementacao.
165 ***************************************************************************
166 """
167 aesgcm = AESGCM(chave)
168 nonce = os.urandom(TAMANHO_NONCE)
169
170 dados_json = json.dumps(dados).encode("utf-8")
171
172 ciphertext = aesgcm.encrypt(
173 nonce,
174 dados_json,
175 None
176 )
177
178 return {
179 "nonce": bytes_para_base64(nonce),
180 "ciphertext": bytes_para_base64(ciphertext)
181 }

184def descriptografar_json(chave: bytes, pacote: dict) -> dict:
185 """
186 ***************************************************************************
187 Funcao: descriptografar_json
188
189 @brief Abre um pacote JSON cifrado por criptografar_json.
190
191 Descricao:
192 Decodifica nonce e ciphertext de Base64, executa a abertura AES-GCM e
193 converte o JSON resultante para dicionario Python. Se o pacote tiver sido
194 adulterado ou a chave estiver incorreta, a biblioteca criptografica falha.
195
196 Parametros:
197 @param chave Chave AES usada para abrir o pacote.
198 @param pacote Dicionario contendo `nonce` e `ciphertext`.
199
200 Valor retornado:
201 @return Dicionario original descriptografado.
202
203 Assertiva de entrada:
204 @pre O pacote deve conter campos `nonce` e `ciphertext` em Base64.
205 @pre A chave deve corresponder a usada na cifragem.
206
207 Assertiva de saida:
208 @post Retorna os dados originais se autenticacao AES-GCM for valida.
209
210 Excecoes:
211 @throws Exception Pode propagar erro de Base64, JSON ou tag AES-GCM invalida.
212
213 Observacoes:
214 Esta funcao e usada para abrir TGTs, Service Tickets, autenticadores,
215 confirmacoes de autenticacao mutua e requisicoes protegidas.
216 ***************************************************************************
217 """
218 aesgcm = AESGCM(chave)
219 nonce = base64_para_bytes(pacote["nonce"])
220 ciphertext = base64_para_bytes(pacote["ciphertext"])
221
222 dados_json = aesgcm.decrypt(
223 nonce,
224 ciphertext,
225 None
226 )
227
228 return json.loads(dados_json.decode("utf-8"))
