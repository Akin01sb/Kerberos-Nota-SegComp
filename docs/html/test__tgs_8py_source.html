26def montar_tgt(usuario="ana", validade_segundos=600, timestamp_emissao=None):
27 """Monta um TGT cifrado para cenarios de teste do TGS."""
28 chave_sessao = gerar_chave_simetrica()
29 chave_sessao_base64 = bytes_para_base64(chave_sessao)
30
31 tgt = criar_tgt(
32 id_cliente=usuario,
33 chave_sessao_cliente_tgs_base64=chave_sessao_base64
34 )
35 tgt["validade_segundos"] = validade_segundos
36
37 if timestamp_emissao is not None:
38 tgt["timestamp_emissao"] = timestamp_emissao
39
40 return chave_sessao_base64, criptografar_json(CHAVE_SECRETA_TGS, tgt)
