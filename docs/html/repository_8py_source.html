22def _normalizar_dados(dados):
23 """
24 @brief Garante que a estrutura raiz de notas esteja no formato esperado.
25
26 @param dados Conteudo carregado do JSON.
27 @return Dicionario com a chave `notas`.
28 """
29 if not isinstance(dados, dict):
30 return {"notas": {}}
31
32 if "notas" not in dados:
33 return {"notas": {}}
34
35 if not isinstance(dados["notas"], dict):
36 return {"notas": {}}
37
38 return dados
